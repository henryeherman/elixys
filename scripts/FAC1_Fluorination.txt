// This will take the FAC synthesis through fluorination.  Make sure you've done everything in FAC0_Setup.txt before starting

// Initialize and make sure robots home
Init()
HomeRobots()

// Move reactor 1 to Add position
Move(Reactor1,Add)

// Set stopcock positions for F-18 trap
ReactorStopcockPosition(1,1,2)
ReactorStopcockPosition(1,2,1)

// Load F-18
//SetPressureRegulator(2,1)
//LoadF18Start()
//RampPressure(2,5,60)
//LoadF18Stop()

// Elute F-18
ReactorStopcockPosition(1,1,1)
ReactorStopcockPosition(1,2,2)
SetPressureRegulator(2,1)
EluteF18Start()
RampPressure(2,5,180)
EluteF18Stop()

// Manually swap the eluent vial with a K222/MeCN solution from position 1.1
// Elute again to flush the lines and get any remaining activity into reactor 1
SetPressureRegulator(2,1)
EluteF18Start()
RampPressure(2,5,60)
EluteF18Stop()

// Take activity measurement
Move(Reactor1,Install)

// Mix and evaporate
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)
Evaporate(Reactor1,110,480,70,450)

// MeCN addition #1, mix and evaporate
Add(Reactor1,Reactor1,2,2)
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)
Evaporate(Reactor1,110,180,70,450)

// MeCN addition #2, mix and evaporate
Add(Reactor1,Reactor1,3,2)
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)
Evaporate(Reactor1,110,180,70,450)

// MeCN addition #3, mix and evaporate
Add(Reactor1,Reactor1,4,2)
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)
Evaporate(Reactor1,110,180,70,450)

// MeCN addition #4, mix and evaporate
Add(Reactor1,Reactor1,5,2)
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)
Evaporate(Reactor1,110,180,50,450)

// Manually add the first precursor and mix
Move(Reactor1,Install)
SetMotorSpeed(1,450)
SetMotorSpeed(1,0)

// Fluorination
React(Reactor1,155,900,50,React1,450)

// Activity measurement
Move(Reactor1,Install)
